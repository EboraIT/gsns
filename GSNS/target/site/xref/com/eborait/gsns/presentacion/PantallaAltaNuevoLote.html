133 	/**
134 	 * Invoca al gestor para realizar el alta del lote de vacunas.
135 	 * 
136 	 * @param frame JFrame de la aplicación.
137 	 */
138 	private void registrarLote(Main frame) {
139 		if (validar()) {
140 			try {
141 				boolean correcto = frame.getGestorGSNS().getGestorRepartoVacunas().altaNuevoLoteVacunas(
142 						Integer.parseInt(lblIdLote.getText()), txtFechaAlta.getText(), Integer.parseInt(txtCantidad.getText()),
143 						txtNombreVacuna.getText(), txtFarmaceutica.getText(), txtFechaAprobacion.getText());
144 				if (correcto) {
145 					JOptionPane.showMessageDialog(frame, "El nuevo lote se ha registrado correctamente.", Main.INFO,
146 							JOptionPane.INFORMATION_MESSAGE);
147 					frame.cambiarPanel(frame.getPanelMain());
148 				}
149 			} catch (NumberFormatException nfe) {
150 				JOptionPane.showMessageDialog(frame,
151 						"Se ha producido un error al registrar el alta del nuevo lote: La cantidad no es correcta. Introduce un número entero.",
152 						Main.ERROR, JOptionPane.ERROR_MESSAGE);
153 			} catch (GSNSException gsnse) {
154 				JOptionPane.showMessageDialog(frame, gsnse.getMessage(), Main.ERROR, JOptionPane.ERROR_MESSAGE);
155 			}
156 		} else {
157 			JOptionPane.showMessageDialog(frame, "Rellena todos los campos.", Main.WARNING,
158 					JOptionPane.WARNING_MESSAGE);
159 		}
160 	}
