142 	/**
143 	 * Invoca al gestor para realizar el alta de entrega de vacunas.
144 	 * 
145 	 * @param frame JFrame de la aplicación.
146 	 */
147 	private void registrarAlta(Main frame) {
148 		if (validar()) {
149 			try {
150 				boolean correcto = frame.getGestorGSNS().getGestorVacunacion().altaEntregaVacunas(
151 						txtIdEntrega.getText(), Integer.parseInt(txtLote.getText()), txtFecha.getText(),
152 						Integer.parseInt(txtCantidad.getText()), comboGrupoPrioridad.getSelectedIndex() + 1,
153 						comboTipoVacuna.getSelectedItem().toString(), comboRegion.getSelectedIndex() + 1);
154 				if (correcto) {
155 					JOptionPane.showMessageDialog(frame, "La entrega se ha registrado correctamente.", Main.INFO,
156 							JOptionPane.INFORMATION_MESSAGE);
157 					frame.cambiarPanel(frame.getPanelMain());
158 				}
159 			} catch (NumberFormatException nfe) {
160 				JOptionPane.showMessageDialog(frame,
161 						"Se ha producido un error al registrar el alta de la entrega de vacunas: La cantidad no es correcta. Introduce un número entero.",
162 						Main.ERROR, JOptionPane.ERROR_MESSAGE);
163 			} catch (GSNSException gsnse) {
164 				JOptionPane.showMessageDialog(frame, gsnse.getMessage(), Main.ERROR, JOptionPane.ERROR_MESSAGE);
165 			}
166 		} else {
167 			JOptionPane.showMessageDialog(frame, "Rellena todos los campos.", Main.WARNING,
168 					JOptionPane.WARNING_MESSAGE);
169 		}
170 	}

172 	/**

177 	private boolean validar() {
178 		JTextField[] textFields = { txtIdEntrega, txtLote, txtFecha, txtCantidad };
179 		for (JTextField jTextField : textFields) {
180 			if (jTextField.getText().length() == 0)

182 		}
183 		return !(comboGrupoPrioridad.getSelectedIndex() == -1 || comboRegion.getSelectedIndex() == -1
184 				|| comboTipoVacuna.getSelectedItem() == null);
