141 	/**
142 	 * Invoca al gestor para registrar la vacunación.
143 	 * 
144 	 * @param frame JFrame de la aplicación.
145 	 */
146 	private void registrarVacunacion(Main frame) {
147 		if (validar()) {
148 			try {
149 				boolean correcto = frame.getGestorGSNS().getGestorVacunacion().registrarVacunacion(txtFecha.getText(),
150 						txtNombre.getText(), txtApellidos.getText(), txtDni.getText(),
151 						comboTipoVacuna.getSelectedItem().toString(), comboGrupoPrioridad.getSelectedIndex() + 1,
152 						comboRegion.getSelectedIndex() + 1, chkSegundaDosis.isSelected());
153 				if (correcto) {
154 					JOptionPane.showMessageDialog(frame, "La vacunación se ha registrado correctamente.", Main.INFO,
155 							JOptionPane.INFORMATION_MESSAGE);
156 					frame.cambiarPanel(frame.getPanelMain());
157 				}
158 			} catch (GSNSException gsnse) {
159 				JOptionPane.showMessageDialog(frame, gsnse.getMessage(), Main.ERROR, JOptionPane.ERROR_MESSAGE);
160 			}
161 		} else {
162 			JOptionPane.showMessageDialog(frame, "Rellena todos los campos.", Main.WARNING,
163 					JOptionPane.WARNING_MESSAGE);
164 		}
165 	}

167 	/**
168 	 * Valida los campos de texto y desplegables.
169 	 * 
170 	 * @return true si la validación es correcta, false de lo contrario.
171 	 */
172 	private boolean validar() {
173 		JTextField[] textFields = { txtNombre, txtApellidos, txtFecha, txtDni };
174 		for (JTextField jTextField : textFields) {
175 			if (jTextField.getText().length() == 0)
176 				return false;
177 		}
178 		return !(comboGrupoPrioridad.getSelectedIndex() == -1 || comboRegion.getSelectedIndex() == -1
179 				|| comboTipoVacuna.getSelectedItem() == null);
180 	}
